#Two sums solution

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {

        vector<int> position;
        for(int i=0; i<nums.size();i++){
            for(int j=i+1;j<nums.size();j++){
                if(nums[i]+nums[j]==target&&i!=j){
                    position ={i,j};
                    return position;
                }
            }
        }
        return position;
    }
};
